148  std::shared_ptr<T> getChildByIDAndType(const std::string& componentID) {
149  std::vector<std::shared_ptr<BaseComponent>> childVector;
150  std::vector<std::shared_ptr<BaseComponent>> grandChildVector = children;
151  bool repeat = true;
152  while (repeat) {
153  childVector = grandChildVector;
154  grandChildVector.clear();
155  repeat = false;
156  for (const auto& child: childVector) {
157  if (child->id == componentID) {
158  if (auto ptr = std::dynamic_pointer_cast<T>(child)) {
159  return ptr;
160  }
161  }
162  const auto& grandchildren = child->getChildren();
163  if (!grandchildren.empty()) {
164  repeat = true;
165  grandChildVector.insert(grandChildVector.end(), grandchildren.begin(), grandchildren.end());
166  }
167  }
168  }
169  return std::shared_ptr<T>();
170  }
